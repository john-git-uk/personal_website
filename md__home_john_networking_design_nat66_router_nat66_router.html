NAT66 Router

This alpine router vm uses NAT on 2001:db8 to allow lab nodes to access the internet.

APK

The ip6tables will perform the NAT translation. Conntrack-tools and tcpdump are useful tools for testing.

apk add iproute2
apk add ip6tables
apk add conntrack-tools
apk add tcpdump

Enable and Disable IPv6 Quick

These are just some scripts to help with testing

/root/ipv6.sh

#!/bin/ash
# Enable IPv6 for all interfaces
echo "Enabling IPv6..."
sysctl -w net.ipv6.conf.all.disable_ipv6=0
sysctl -w net.ipv6.conf.default.disable_ipv6=0
sysctl -w net.ipv6.conf.lo.disable_ipv6=0

/root/noipv6.sh

#!/bin/ash
# Disable IPv6 for all interfaces
sysctl -w net.ipv6.conf.all.disable_ipv6=1
sysctl -w net.ipv6.conf.default.disable_ipv6=1
sysctl -w net.ipv6.conf.lo.disable_ipv6=1

sysctl Settings

These settings control ipv6 routing and whether the system uses the routing advertisments to configure addresses.

echo "net.ipv6.conf.all.accept_ra = 0" | sudo tee -a /etc/sysctl.conf
echo "net.ipv6.conf.default.accept_ra = 0" | sudo tee -a /etc/sysctl.conf
echo "net.ipv6.conf.eth_int0.accept_ra = 0" | tee -a /etc/sysctl.conf
echo "net.ipv6.conf.eth_out0.accept_ra = 0" | tee -a /etc/sysctl.conf
echo "net.ipv6.conf.all.forwarding = 1" | tee -a /etc/sysctl.conf

Assigning IP to Outgoing Interface

This uses the ip command from the iproute2 package to dynamicly add all of the inside global addresses to the WAN facing interface.

/root/assign_nat_pool.sh

REDACTED must be replaced with your valid IPv6 globally routable prefix.

#!/bin/ash
# Interface name
INTERFACE="eth_out0"
# Base prefix
BASE="REDACTED:ffff:ffff:ff00:"
# Assign addresses using a while loop
i=0
while [ $i -le 255 ]; do
SUFFIX=$(printf "%02x" $i)
ip -6 addr add ${BASE}${SUFFIX}/120 dev $INTERFACE
i=$((i + 1))
done

Disabling Bootloader Network Names

Create a service to automatically generate route to any connected link local

#!/bin/ash
echo '#!/bin/ash
# Configuration
interface="eth_int0"
route_prefix="2001:db8::/28"
current_neighbor=""
while true; do
# Get the link-local address of the interface
own_link_local=$(ip -6 addr show dev $interface | grep "scope link" | awk '{print $2}' | cut -d'/' -f1)
# Get the first neighbor link-local address that is not the interface's own address
new_neighbor=$(ip -6 neigh show dev $interface | grep "fe80::" | awk '{print $1}' | grep -v "$own_link_local" | head -n 1)
# Check if the neighbor address has changed
if [ "$new_neighbor" != "$current_neighbor" ]; then
# Delete the old route if it exists and if current_neighbor is not null
if [ -n "$current_neighbor" ]; then
ip -6 route del $route_prefix via $current_neighbor dev $interface
logger -p user.notice "autoroute: Old route deleted via $current_neighbor"
fi
# Add the new route if a valid neighbor address was found and new_neighbor is not null
if [ -n "$new_neighbor" ]; then
ip -6 route add $route_prefix via $new_neighbor dev $interface
logger -p user.notice "autoroute: Route added via $new_neighbor"
current_neighbor=$new_neighbor
else
logger -p user.notice "autoroute: No valid neighbor link-local address found"
current_neighbor=""
fi
fi
# Wait for 1 second before checking again
sleep 1
done
' > /root/autoroute.sh
chmod +x /root/autoroute.sh
echo '#!/sbin/openrc-run
description="Monitor and Update IPv6 Route for eth_int0"
depend() {
need net
}
start() {
ebegin "Starting autoroute"
start-stop-daemon --start --exec /root/autoroute.sh --background
eend $?
}
stop() {
ebegin "Stopping autoroute"
start-stop-daemon --stop --exec /root/autoroute.sh
eend $?
}
' > /etc/init.d/autoroute
chmod +x /etc/init.d/autoroute
# Add the service to the default runlevel
rc-update add autoroute default
rc-service autoroute start

Ash Commands to Enable Nat

REDACTED must be replaced with your valid IPv6 globally routable prefix.

apk add ip6tables
apk add conntrack-tools
# Flush all existing rules
ip6tables -F
ip6tables -t nat -F
ip6tables -X
ip6tables -t nat -X
# Set default policies to ACCEPT
ip6tables -P INPUT ACCEPT
ip6tables -P FORWARD ACCEPT
ip6tables -P OUTPUT ACCEPT
# Allow all traffic explicitly
ip6tables -A INPUT -j ACCEPT
ip6tables -A FORWARD -j ACCEPT
ip6tables -A OUTPUT -j ACCEPT
# Add the NAT66 rule
ip6tables -t nat -A POSTROUTING -o eth_out0 -j NETMAP --to REDACTED:ffff:ffff:ff00::/120
# Exclude the main address from being natted
ip6tables -t nat -I POSTROUTING -o eth_out0 -s REDACTED::2 -j RETURN
# Save and reload on reboot
ip6tables-save > /etc/ip6tables.rules
mkdir -p /etc/network/if-pre-up.d
echo '#!/bin/sh' > /etc/network/if-pre-up.d/ip6tables
echo 'ip6tables-restore < /etc/ip6tables.rules' >> /etc/network/if-pre-up.d/ip6tables
chmod +x /etc/network/if-pre-up.d/ip6tables
